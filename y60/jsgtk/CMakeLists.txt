    DEVELOPMENT_INSTALL_COMPONENT G60_development

    SOVERSION ${VERSION_MAJOR}

# fix OpenGL/X11 link cycle problem
if(OSX)
    get_property(FLAGS TARGET y60jsgtk PROPERTY LINK_FLAGS)
    set_property(TARGET y60jsgtk PROPERTY LINK_FLAGS "${FLAGS} -Wl,-dylib_file,/System/Library/Frameworks/OpenGL.framework/Versions/A/Libraries/libGL.dylib:/System/Library/Frameworks/OpenGL.framework/Versions/A/Libraries/libGL.dylib")
    set(FLAGS)
endif(OSX)
